% Safety net for images with no explicit width in the markdown. We want
% two behaviours from the same default:
%   - small toolbar icons (<< linewidth natural size) → render at natural DPI
%   - large screenshots (>> linewidth natural size) → clamp to linewidth
% The standard LaTeX idiom for this (and what Pandoc's own default preamble
% ships) is to compute \maxwidth / \maxheight per-image at include time,
% using \ifdim to pick the smaller of natural size and the page limit.
%
% Gin's "max width" key does NOT exist in vanilla graphicx, and
% \setkeys{Gin}{max width=\linewidth} is silently ignored — confirmed
% empirically by pdfimages showing xppi=72 (no clamp fired) on 1600-px
% screenshots. The \maxwidth approach works because "width" IS a real
% Gin key and \ifdim evaluates per inclusion.
%
% This also matters because Pandoc silently drops attribute values that
% don't match its numeric-unit grammar (e.g. width=.6cm, width=\textwidth,
% width=0.8\textwidth). When that happens the Gin default is all we have.
\makeatletter
\def\maxwidth{\ifdim\Gin@nat@width>\linewidth\linewidth\else\Gin@nat@width\fi}
\def\maxheight{\ifdim\Gin@nat@height>0.9\textheight 0.9\textheight\else\Gin@nat@height\fi}
\makeatother
\setkeys{Gin}{width=\maxwidth,height=\maxheight,keepaspectratio}

% Cover page. Layout mirrors the engine-reference source: a three-line
% Huge blue title block (product name, then two optional continuation
% lines from manuals.yaml), a 5-baseline gap, then a LARGE black
% "coverKind" line (e.g. Reference Manual / Tutorials), the date, and a
% big vertical skip down to the Large blue publisher line at the bottom.
% When coverKind is absent the template falls back to the existing
% subtitle-in-blue layout used by qgis-reference and tutorials.

\begingroup
\thispagestyle{empty}%
\hbox{%
  \hspace*{0.05\textwidth}%
  \rule{1pt}{\textheight}%
  \hspace*{0.05\textwidth}%
  \parbox[b]{0.85\textwidth}{%
    \noindent{\Huge\sffamily\color{hydroniaPrimary} $title$%
    $if(titleLine2)$\\[0.5\baselineskip]$titleLine2$$endif$%
    $if(titleLine3)$\\[0.5\baselineskip]$titleLine3$$endif$\hfill}\\[5\baselineskip]
    $if(coverKind)$%
      {\noindent\LARGE\sffamily\color{black}$coverKind$}\\[1\baselineskip]
    $else$$if(subtitle)$%
      {\noindent\Large\sffamily\color{hydroniaPrimary}$subtitle$}\\[1\baselineskip]
    $endif$$endif$
    {\noindent\large\sffamily\color{black}$date$}
    $if(coverLogo)$\\[1\baselineskip]\includegraphics[height=0.8in]{$coverLogo$}$endif$
